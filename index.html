Exploring Derivatives via Secants

Welcome to my hobby maths site! Below you can visualize how a secant line approaches a tangent line. Use the inputs to change the function or the point of interest.

Note: The first load might take a moment as your browser downloads the Python runtime.

packages = ["manim", "numpy"] from manim import * import numpy as np # This function runs when the user clicks the button def render_scene(event=None): # 1. Grab inputs from HTML user_func_str = Element("func-input").element.value x1_val = float(Element("point-input").element.value) # 2. Define the scene class InteractiveSecant(Scene): def construct(self): axes = Axes(x_range=[-3, 3], y_range=[-2, 9], axis_config={"color": BLUE}) # Safe eval for math functions (using numpy for safety) def f(x): return eval(user_func_str, {"x": x, "np": np}) graph = axes.plot(f, color=WHITE) x2_tracker = ValueTracker(x1_val + 1.5) # The dynamic Secant Line secant_line = always_redraw( lambda: axes.plot( lambda x: ((f(x2_tracker.get_value()) - f(x1_val)) / (x2_tracker.get_value() - x1_val)) * (x - x1_val) + f(x1_val), color=RED, x_range=[x1_val - 2, x1_val + 2] ) ) dot1 = Dot(axes.c2p(x1_val, f(x1_val)), color=YELLOW) dot2 = always_redraw(lambda: Dot(axes.c2p(x2_tracker.get_value(), f(x2_tracker.get_value())), color=RED)) self.add(axes, graph, dot1, dot2) self.play(Create(secant_line)) self.play(x2_tracker.animate.set_value(x1_val + 0.01), run_time=3) self.wait() # 3. Render the scene into the div InteractiveSecant().render()
